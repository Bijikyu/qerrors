/**
 * Shared logging utilities for consistent function entry/exit logging patterns
 * RATIONALE: Replaces debug package with console methods for dependency reduction
 * Maintains consistent logging patterns while using native browser/console APIs
 */
const{safeStringify}=require('./validation.js'),environment=require('./environment'),isDevelopment=environment.isServer&&process.env?.NODE_ENV!=='production';

const logFunctionEntry=(functionName,params=null)=>{if(!isDevelopment)return;const paramStr=params!==null?(typeof params==='string'?(params?.length>100?params.substring(0,97)+'...':params):typeof params==='object'&&params!==null?safeStringify(params):String(params)):null;console.log(`[react-hooks:log] ${functionName} is running${paramStr?` with ${paramStr}`:''}`);};
const logFunctionExit=(functionName,result=null)=>{if(!isDevelopment)return;const resultStr=result!==null?(typeof result==='string'?(result?.length>100?result.substring(0,97)+'...':result):typeof result==='object'&&result!==null?safeStringify(result):String(result)):null;console.log(`[react-hooks:log] ${functionName} is returning${resultStr?` ${resultStr}`:''}`);};
const logError=(functionName,error,context='')=>{if(typeof console!=='undefined')console.error(`[react-hooks:error] ${functionName} error${context?` (${context})`:''}:`,error);};
const logWarning=(functionName,message,data=null)=>{if(!isDevelopment||typeof console==='undefined')return;console.warn(`[react-hooks:warn] ${functionName} warning: ${message}${data!==null?` Data: ${safeStringify(data)}`:''}`);};
const withFunctionLogging=(functionName,fn)=>(...args)=>{logFunctionEntry(functionName,args);try{const result=fn(...args);return result&&typeof result.then==='function'?result.then(asyncResult=>(logFunctionExit(functionName,asyncResult),asyncResult)).catch(error=>{logError(functionName,error,'returned error');throw error;}):(logFunctionExit(functionName,result),result);}catch(error){logError(functionName,error,'threw error');throw error;}};
const createDebugLogger=namespace=>(message,...args)=>{if(isDevelopment&&typeof console!=='undefined')console.log(`[react-hooks:${namespace}] ${message}`,...args);};
module.exports={logFunctionEntry,logFunctionExit,logError,logWarning,withFunctionLogging,safeStringify,createDebugLogger};
